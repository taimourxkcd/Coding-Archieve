CREATE TABLE UserNew
(
  Username_ VARCHAR(30) NOT NULL,
  Password_ VARCHAR(30) NOT NULL,
  DOB DATE NOT NULL,
  USER_Name_ VARCHAR(30) NOT NULL,
  Age INTEGER NOT NULL,
  PRIMARY KEY (Username_)
);

CREATE TABLE User_Email
(
  Email VARCHAR(30) NOT NULL,
  Username_ VARCHAR(30) NOT NULL,
  FOREIGN KEY (Username_) REFERENCES UserNew(Username_)
);

CREATE TABLE User_Address
(
  Address VARCHAR(30) NOT NULL,
  Username_ VARCHAR(30) NOT NULL,
  FOREIGN KEY (Username_) REFERENCES UserNew(Username_)
);

CREATE TABLE Artist
(
  Artist_id INTEGER NOT NULL,
  Artist_Name VARCHAR(30) NOT NULL,
  age INTEGER NOT NULL,
  PRIMARY KEY (Artist_id)
);

CREATE TABLE Artist_Email
(
  Email VARCHAR(30) NOT NULL,
  Artist_id INTEGER NOT NULL,
  FOREIGN KEY (Artist_id) REFERENCES Artist(Artist_id)
);

CREATE TABLE Artist_Social
(
  Instagram VARCHAR(20) NOT NULL,
  facebook VARCHAR(20) NOT NULL,
  Artist_id INT NOT NULL,
  FOREIGN KEY (Artist_id) REFERENCES Artist(Artist_id)
);


CREATE TABLE Album
(
  Album_Name VARCHAR(30) NOT NULL,
  Year INTEGER NOT NULL,
  Genre VARCHAR(30) NOT NULL,
  Format VARCHAR(30) NOT NULL,
  Rating_ FLOAT NOT NULL,
  No_of_Songs INTEGER NOT NULL,
  Artist_id INTEGER NOT NULL,
  PRIMARY KEY (Album_Name, Artist_id),
  FOREIGN KEY (Artist_id) REFERENCES Artist(Artist_id)
);


CREATE TABLE Track
(
  Track_Title VARCHAR(30) NOT NULL,
  Duration INTEGER NOT NULL,
  Release_Date INTEGER NOT NULL,
  Rating FLOAT NOT NULL,
  Track_id INTEGER NOT NULL,
  Album_Name VARCHAR(100),
  Artist_id INTEGER NOT NULL,
  PRIMARY KEY (Track_id),
  FOREIGN KEY (Artist_id) REFERENCES Artist(Artist_id),
  FOREIGN KEY (Album_Name, Artist_id) REFERENCES Album(Album_Name, Artist_id)
);

CREATE TABLE Playlist
(
  Playlist_Title VARCHAR(30) NOT NULL,
  No_of_songs INTEGER NOT NULL,
  playlist_id INTEGER NOT NULL,
  Username_ VARCHAR(30) NOT NULL,
  PRIMARY KEY (playlist_id),
  FOREIGN KEY (Username_) REFERENCES UserNew(Username_)
);

CREATE TABLE Playlist_Container
(
  playlist_id INTEGER NOT NULL,
  Track_id INTEGER NOT NULL,
  FOREIGN KEY (playlist_id) REFERENCES Playlist(playlist_id),
  FOREIGN KEY (Track_id) REFERENCES Track(Track_id)
);

CREATE TABLE Playlist_Tags
(
  Tags VARCHAR(30) NOT NULL,
  playlist_id INTEGER NOT NULL,
  FOREIGN KEY (playlist_id) REFERENCES Playlist(playlist_id)
);
