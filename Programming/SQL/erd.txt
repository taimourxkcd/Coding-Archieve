CREATE TABLE Artist
(
  Artist_id INT NOT NULL,
  Artist_Name VARCHAR NOT NULL,
  age INT NOT NULL,
  PRIMARY KEY (Artist_id)
);

CREATE TABLE User
(
  Username_ INT NOT NULL,
  Password_ VARCHAR NOT NULL,
  DOB DATE NOT NULL,
  USER_Name_ VARCHAR NOT NULL,
  Age INT NOT NULL,
  PRIMARY KEY (Username_)
);

CREATE TABLE Artist_Email
(
  Email VARCHAR NOT NULL,
  Artist_id INT NOT NULL,
  FOREIGN KEY (Artist_id) REFERENCES Artist(Artist_id)
);

CREATE TABLE Artist_Social
(
  Social VARCHAR NOT NULL,
  Artist_id INT NOT NULL,
  FOREIGN KEY (Artist_id) REFERENCES Artist(Artist_id)
);

CREATE TABLE User_Email
(
  Email VARCHAR NOT NULL,
  Username_ INT NOT NULL,
  FOREIGN KEY (Username_) REFERENCES User(Username_)
);

CREATE TABLE User_Address
(
  Address VARCHAR NOT NULL,
  Username_ INT NOT NULL,
  FOREIGN KEY (Username_) REFERENCES User(Username_)
);

CREATE TABLE Album
(
  Album_Name VARCHAR NOT NULL,
  Year INT NOT NULL,
  Title VARCHAR NOT NULL,
  Genre VARCHAR NOT NULL,
  Format VARCHAR NOT NULL,
  Release_Date DATE NOT NULL,
  Rating_ INT NOT NULL,
  No_of_Songs INT NOT NULL,
  Artist_id INT NOT NULL,
  PRIMARY KEY (Album_Name, Artist_id),
  FOREIGN KEY (Artist_id) REFERENCES Artist(Artist_id)
);

CREATE TABLE Track
(
  Track_Title VARCHAR NOT NULL,
  Duration INT NOT NULL,
  Release_Date INT NOT NULL,
  Rating INT NOT NULL,
  Track_id INT NOT NULL,
  Artist_id INT NOT NULL,
  Album_Name VARCHAR NOT NULL,
  Artist_id INT NOT NULL,
  PRIMARY KEY (Track_id),
  FOREIGN KEY (Artist_id) REFERENCES Artist(Artist_id),
  FOREIGN KEY (Album_Name, Artist_id) REFERENCES Album(Album_Name, Artist_id)
);

CREATE TABLE Playlist
(
  Playlist_Title VARCHAR NOT NULL,
  No_of_songs INT NOT NULL,
  playlist_id INT NOT NULL,
  Username_ INT NOT NULL,
  PRIMARY KEY (playlist_id),
  FOREIGN KEY (Username_) REFERENCES User(Username_)
);

CREATE TABLE Playlist_Container
(
  playlist_id INT NOT NULL,
  Track_id INT NOT NULL,
  FOREIGN KEY (playlist_id) REFERENCES Playlist(playlist_id),
  FOREIGN KEY (Track_id) REFERENCES Track(Track_id)
);

CREATE TABLE Playlist_Tags
(
  Tags VARCHAR NOT NULL,
  playlist_id INT NOT NULL,
  FOREIGN KEY (playlist_id) REFERENCES Playlist(playlist_id)
);
